### RIVALS GETTING TOO CLOSE - Trigger Special Project
country_event = {
	id = XL_MAP_mission.0001
	title = "XL_MAP_mission.0001.name"
	desc = "XL_MAP_mission.0001.desc"
	picture = GFX_evt_XL_MAP_enterprise_c_into_rift
	show_sound = event_the_pirate_rebellion
	location = event_target:dewa_star

	is_triggered_only = yes

	immediate = {
	}

	### Accept Mission
	option = {
		name = XL_MAP_mission.0001.a
		#custom_tooltip = XL_MAP_mission.0001.a.tt

		enable_special_project = {
			name = "RIVALS_GETTING_TOO_CLOSE"
			location = event_target:dewa_star
			owner = event_target:romulan_country
		}
	}

	### Decline Mission
	option = {
		name = XL_MAP_mission.0001.b
		#custom_tooltip = XL_MAP_mission.0001.b.tt
	}
}
