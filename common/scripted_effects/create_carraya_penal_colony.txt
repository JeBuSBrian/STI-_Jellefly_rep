create_carraya_penal_colony = {
	random_system = {
		limit = {
			has_star_flag = carraya
		}

		create_starbase = {
			size = starbase_outpost
			owner = event_target:romulan_country
		}

		random_system_planet = {
			limit = {
				has_planet_flag = planet_carraya
			}

			if = {
				limit = {
					is_colony = no
				}

				set_owner = event_target:romulan_country
				set_colony_type = col_default

				add_building = building_capital

				while = {
					count = 0
					add_district_and_planet_size_if_needed_effect = {
						district = district_city
					}
				}
				while = {
					count = 0
					add_district_and_planet_size_if_needed_effect = {
						district = district_generator
					}
				}
				while = {
					count = 0
					add_district_and_planet_size_if_needed_effect = {
						district = district_industrial
					}
				}
				while = {
					count = 0
					add_district_and_planet_size_if_needed_effect = {
						district = district_mining
					}
				}
				while = {
					count = 0
					add_district_and_planet_size_if_needed_effect = {
						district = district_farming
					}
				}

				create_pop = {
					species = "Romulan"
				}
				create_pop = {
					species = "Klingon"
				}
			}
			else = {
				create_pop = {
					species = "Klingon"
				}
			}
		}
	}
}
